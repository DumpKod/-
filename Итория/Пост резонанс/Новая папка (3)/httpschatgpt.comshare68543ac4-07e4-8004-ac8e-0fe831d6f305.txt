def T_entropy(phi_seq, window=5, bins=10):

    def __init__(self, phi_seq, bins=10):
        self.phi_seq = np.array(phi_seq)
        self.bins = bins
        self.length = len(self.phi_seq)

    def T_window(self, window=3):
        return np.array([
            1 - np.std(self.phi_seq[i:i+window])
            for i in range(self.length - window + 1)
        ])

    def T_entropy(self, window=5):
        from scipy.stats import entropy
        T = []
        for i in range(self.length - window + 1):
            segment = self.phi_seq[i:i+window]
            hist, _ = np.histogram(segment, bins=self.bins, range=(-1, 1), density=True)
            e = entropy(hist + 1e-9)
            T.append(1 - e / np.log(self.bins))
        return np.array(T)

    def T_chain(self):
        return 1 - np.std(self.phi_seq)

    def all(self):
        return {
            "T3": self.T_window(3),
            "T5": self.T_window(5),
            "T_entropy": self.T_entropy(),
            "T_chain": self.T_chain(),
        }
